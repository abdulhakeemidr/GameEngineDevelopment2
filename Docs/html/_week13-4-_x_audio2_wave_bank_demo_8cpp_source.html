34int main()
35{
36 //
37 // Initialize XAudio2
38 //
39 HRESULT hr = CoInitializeEx(nullptr, COINIT_MULTITHREADED);
40 if (FAILED(hr))
41 {
42 wprintf(L"Failed to init COM: %#X\n", hr);
43 return 0;
44 }
45
46#ifdef USING_XAUDIO2_7_DIRECTX
47 // Workaround for XAudio 2.7 known issue
48#ifdef _DEBUG
49 HMODULE mXAudioDLL = LoadLibraryExW(L"XAudioD2_7.DLL", nullptr, 0x00000800 /* LOAD_LIBRARY_SEARCH_SYSTEM32 */);
50#else
51 HMODULE mXAudioDLL = LoadLibraryExW(L"XAudio2_7.DLL", nullptr, 0x00000800 /* LOAD_LIBRARY_SEARCH_SYSTEM32 */);
52#endif
53 if (!mXAudioDLL)
54 {
55 wprintf(L"Failed to find XAudio 2.7 DLL");
56 CoUninitialize();
57 return 0;
58 }
59#endif
60
61 UINT32 flags = 0;
62#if defined(USING_XAUDIO2_7_DIRECTX) && defined(_DEBUG)
63 flags |= XAUDIO2_DEBUG_ENGINE;
64#endif
65 ComPtr<IXAudio2> pXAudio2;
66 hr = XAudio2Create(pXAudio2.GetAddressOf(), flags);
67 if (FAILED(hr))
68 {
69 wprintf(L"Failed to init XAudio2 engine: %#X\n", hr);
70 CoUninitialize();
71 return 0;
72 }
73
74#if !defined(USING_XAUDIO2_7_DIRECTX) && defined(_DEBUG)
75 // To see the trace output, you need to view ETW logs for this application:
76 // Go to Control Panel, Administrative Tools, Event Viewer.
77 // View->Show Analytic and Debug Logs.
78 // Applications and Services Logs / Microsoft / Windows / XAudio2.
79 // Right click on Microsoft Windows XAudio2 debug logging, Properties, then Enable Logging, and hit OK
80 XAUDIO2_DEBUG_CONFIGURATION debug = { 0 };
81 debug.TraceMask = XAUDIO2_LOG_ERRORS | XAUDIO2_LOG_WARNINGS;
82 debug.BreakMask = XAUDIO2_LOG_ERRORS;
83 pXAudio2->SetDebugConfiguration(&debug, 0);
84#endif
85
86 //
87 // Create a mastering voice
88 //
89 IXAudio2MasteringVoice* pMasteringVoice = nullptr;
90
91 if (FAILED(hr = pXAudio2->CreateMasteringVoice(&pMasteringVoice)))
92 {
93 wprintf(L"Failed creating mastering voice: %#X\n", hr);
94 pXAudio2.Reset();
95 CoUninitialize();
96 return 0;
97 }
98
99 //
100 // Find our wave bank file
101 //
102 WCHAR wavebank[MAX_PATH];
103#if defined(USE_XWMA)
104#if defined(USING_XAUDIO2_8)
105#error xWMA is not supported by XAudio 2.8 on Windows 8.x
106#endif
107
108 if (FAILED(hr = FindMediaFileCch(wavebank, MAX_PATH, L"Media\\Banks\\XWMAdroid.xwb")))
109#elif defined(USE_ADPCM)
110 if (FAILED(hr = FindMediaFileCch(wavebank, MAX_PATH, L"Media\\Banks\\ADPCMdroid.xwb")))
111#else
112 if (FAILED(hr = FindMediaFileCch(wavebank, MAX_PATH, L"Media\\Banks\\droid.xwb")))
113#endif
114 {
115 wprintf(L"Failed to find media file (%#X)\n", hr);
116 pXAudio2.Reset();
117 CoUninitialize();
118 return 0;
119 }
120
121 //
122 // Extract wavebank data (entries, formats, offsets, and sizes)
123 //
124 WaveBankReader wb;
125
126 if (FAILED(hr = wb.Open(wavebank)))
127 {
128 wprintf(L"Failed to wavebank data (%#X)\n", hr);
129 pXAudio2.Reset();
130 CoUninitialize();
131 return 0;
132 }
133
134 wprintf(L"Wavebank loaded with %u entries.\n", wb.Count());
135
136 if (wb.IsStreamingBank())
137 {
138 wprintf(L"This sample plays back in-memory wave banks.\nSee XAudio2AsyncStream for playing streaming wave banks");
139 pXAudio2.Reset();
140 CoUninitialize();
141 return 0;
142 }
143
144 //
145 // At this point, the Wave Bank's metadata, format, entries, and names (if present) are available
146 // The actual wave data for an in-memory buffer is potentially still loading asynchronously.
147 // (note a streaming wave bank is prepared as soon as Open returns)
148 //
149 // We can either call WaitOnPrepare() to synchronously wait for this load, -or- we can
150 // routinely check wb.IsPrepared() to see if it's ready yet...
151 //
152 wb.WaitOnPrepare();
153
154 //
155 // Play sounds from wave bank
156 //
157 for (uint32_t j = 0; j < wb.Count(); ++j)
158 {
159 WaveBankReader::Metadata metadata;
160 if (FAILED(hr = wb.GetMetadata(j, metadata)))
161 {
162 wprintf(L"Failed getting metadata for index %u: %#X\n", j, hr);
163 pXAudio2.Reset();
164 CoUninitialize();
165 return 0;
166 }
167
168 if (metadata.loopLength > 0 && metadata.loopLength != metadata.duration)
169 wprintf(L"Playing entry %u (duration of %u samples; loop point %u,%u )...\n", j, metadata.duration, metadata.loopStart, metadata.loopLength);
170 else
171 wprintf(L"Playing entry %u (duration of %u samples)...\n", j, metadata.duration);
172
173 if (FAILED(hr = PlayWaveFromWaveBank(pXAudio2.Get(), wb, j)))
174 {
175 wprintf(L"Failed creating source voice for index %u: %#X\n", j, hr);
176 pXAudio2.Reset();
177 CoUninitialize();
178 return 0;
179 }
180 }
181
182 //
183 // Cleanup XAudio2
184 //
185
186 // All XAudio2 interfaces are released when the engine is destroyed, but being tidy
187 pMasteringVoice->DestroyVoice();
188
189 pXAudio2.Reset();
190
191#ifdef USING_XAUDIO2_7_DIRECTX
192 if (mXAudioDLL)
193 FreeLibrary(mXAudioDLL);
194#endif
195
196 CoUninitialize();
197}

200//--------------------------------------------------------------------------------------
201// Name: PlayWaveFromWaveBank
202// Desc: Plays a wave and blocks until the wave finishes playing
203//--------------------------------------------------------------------------------------
204_Use_decl_annotations_
205HRESULT PlayWaveFromWaveBank(IXAudio2* pXaudio2, WaveBankReader& wb, uint32_t index)
206{
207 if (index >= wb.Count())
208 return E_INVALIDARG;
209
210 uint8_t waveFormat[64];
211 auto pwfx = reinterpret_cast<WAVEFORMATEX*>(&waveFormat);
212
213 HRESULT hr = wb.GetFormat(index, pwfx, 64);
214 if (FAILED(hr))
215 return hr;
216
217 const uint8_t* waveData = nullptr;
218 uint32_t waveSize;
219
220 hr = wb.GetWaveData(index, &waveData, waveSize);
221 if (FAILED(hr))
222 return hr;
223
224 WaveBankReader::Metadata metadata;
225 hr = wb.GetMetadata(index, metadata);
226 if (FAILED(hr))
227 return hr;
228
229 //
230 // Play the wave using a XAudio2SourceVoice
231 //
232
233 // Create the source voice
234 IXAudio2SourceVoice* pSourceVoice;
235 if (FAILED(hr = pXaudio2->CreateSourceVoice(&pSourceVoice, pwfx)))
236 {
237 wprintf(L"Error %#X creating source voice\n", hr);
238 return hr;
239 }
240
241 // Submit the wave sample data using an XAUDIO2_BUFFER structure
242 XAUDIO2_BUFFER buffer = { 0 };
243 buffer.pAudioData = waveData;
244 buffer.Flags = XAUDIO2_END_OF_STREAM; // tell the source voice not to expect any data after this buffer
245 buffer.AudioBytes = waveSize;
246
247 if (metadata.loopLength > 0 && metadata.loopLength != metadata.duration)
248 {
249 buffer.LoopBegin = metadata.loopStart;
250 buffer.LoopLength = metadata.loopLength;
251 buffer.LoopCount = 1; // We'll just assume we play the loop twice
252 }
253
254 const uint32_t* seekTable;
255 uint32_t seekTableCount;
256 uint32_t tag;
257 hr = wb.GetSeekTable(index, &seekTable, seekTableCount, tag);
258
259 if (seekTable)
260 {
261 if (tag == WAVE_FORMAT_WMAUDIO2 || tag == WAVE_FORMAT_WMAUDIO3)
262 {
263#if defined(USING_XAUDIO2_7_DIRECTX) || defined(USING_XAUDIO2_9)
264 XAUDIO2_BUFFER_WMA xwmaBuffer = { 0 };
265 xwmaBuffer.pDecodedPacketCumulativeBytes = seekTable;
266 xwmaBuffer.PacketCount = seekTableCount;
267 if (FAILED(hr = pSourceVoice->SubmitSourceBuffer(&buffer, &xwmaBuffer)))
268 {
269 wprintf(L"Error %#X submitting source buffer (xWMA)\n", hr);
270 pSourceVoice->DestroyVoice();
271 return hr;
272 }
273#else
274 wprintf(L"This platform does not support xWMA\n");
275 pSourceVoice->DestroyVoice();
276 return hr;
277#endif
278 }
279 else if (tag == 0x166 /* WAVE_FORMAT_XMA2 */)
280 {
281 wprintf(L"This platform does not support XMA2\n");
282 pSourceVoice->DestroyVoice();
283 return hr;
284 }
285 }
286 else if (FAILED(hr = pSourceVoice->SubmitSourceBuffer(&buffer)))
287 {
288 wprintf(L"Error %#X submitting source buffer\n", hr);
289 pSourceVoice->DestroyVoice();
290 return hr;
291 }
292
293 hr = pSourceVoice->Start(0);
294
295 // Let the sound play
296 BOOL isRunning = TRUE;
297 while (SUCCEEDED(hr) && isRunning)
298 {
299 XAUDIO2_VOICE_STATE state;
300 pSourceVoice->GetState(&state);
301 isRunning = (state.BuffersQueued > 0) != 0;
302
303 // Wait till the escape key is pressed
304 if (GetAsyncKeyState(VK_ESCAPE))
305 break;
306
307 Sleep(10);
308 }
309
310 // Wait till the escape key is released
311 while (GetAsyncKeyState(VK_ESCAPE))
312 Sleep(10);
313
314 pSourceVoice->DestroyVoice();
315
316 return hr;
317}

323_Use_decl_annotations_
324HRESULT FindMediaFileCch(WCHAR* strDestPath, int cchDest, LPCWSTR strFilename)
325{
326 bool bFound = false;
327
328 if (!strFilename || strFilename[0] == 0 || !strDestPath || cchDest < 10)
329 return E_INVALIDARG;
330
331 // Get the exe name, and exe path
332 WCHAR strExePath[MAX_PATH] = { 0 };
333 WCHAR strExeName[MAX_PATH] = { 0 };
334 WCHAR* strLastSlash = nullptr;
335 GetModuleFileName(nullptr, strExePath, MAX_PATH);
336 strExePath[MAX_PATH - 1] = 0;
337 strLastSlash = wcsrchr(strExePath, TEXT('\\'));
338 if (strLastSlash)
339 {
340 wcscpy_s(strExeName, MAX_PATH, &strLastSlash[1]);
341
342 // Chop the exe name from the exe path
343 *strLastSlash = 0;
344
345 // Chop the .exe from the exe name
346 strLastSlash = wcsrchr(strExeName, TEXT('.'));
347 if (strLastSlash)
348 *strLastSlash = 0;
349 }
350
351 wcscpy_s(strDestPath, cchDest, strFilename);
352 if (GetFileAttributes(strDestPath) != 0xFFFFFFFF)
353 return S_OK;
354
355 // Search all parent directories starting at .\ and using strFilename as the leaf name
356 WCHAR strLeafName[MAX_PATH] = { 0 };
357 wcscpy_s(strLeafName, MAX_PATH, strFilename);
358
359 WCHAR strFullPath[MAX_PATH] = { 0 };
360 WCHAR strFullFileName[MAX_PATH] = { 0 };
361 WCHAR strSearch[MAX_PATH] = { 0 };
362 WCHAR* strFilePart = nullptr;
363
364 GetFullPathName(L".", MAX_PATH, strFullPath, &strFilePart);
365 if (!strFilePart)
366 return E_FAIL;
367
368 while (strFilePart && *strFilePart != '\0')
369 {
370 swprintf_s(strFullFileName, MAX_PATH, L"%s\\%s", strFullPath, strLeafName);
371 if (GetFileAttributes(strFullFileName) != 0xFFFFFFFF)
372 {
373 wcscpy_s(strDestPath, cchDest, strFullFileName);
374 bFound = true;
375 break;
376 }
377
378 swprintf_s(strFullFileName, MAX_PATH, L"%s\\%s\\%s", strFullPath, strExeName, strLeafName);
379 if (GetFileAttributes(strFullFileName) != 0xFFFFFFFF)
380 {
381 wcscpy_s(strDestPath, cchDest, strFullFileName);
382 bFound = true;
383 break;
384 }
385
386 swprintf_s(strSearch, MAX_PATH, L"%s\\..", strFullPath);
387 GetFullPathName(strSearch, MAX_PATH, strFullPath, &strFilePart);
388 }
389 if (bFound)
390 return S_OK;
391
392 // On failure, return the file as the path but also return an error code
393 wcscpy_s(strDestPath, cchDest, strFilename);
394
395 return HRESULT_FROM_WIN32(ERROR_FILE_NOT_FOUND);
396}
