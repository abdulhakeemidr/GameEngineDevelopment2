115bool BoxApp::Initialize()
116{
117 if(!D3DApp::Initialize())
118 return false;
119
120 // Reset the command list to prep for initialization commands.
121 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), nullptr));
122
123 BuildDescriptorHeaps();
124 BuildConstantBuffers();
125 BuildRootSignature();
126 BuildShadersAndInputLayout();
127 BuildBoxGeometry();
128 BuildPSO();
129
130 // Execute the initialization commands.
131 ThrowIfFailed(mCommandList->Close());
132 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
133 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
134
135 // Wait until initialization is complete.
136 FlushCommandQueue();
137
138 return true;
139}

152void BoxApp::Update(const GameTimer& gt)
153{
154 // Convert Spherical to Cartesian coordinates.
155 float x = mRadius*sinf(mPhi)*cosf(mTheta);
156 float z = mRadius*sinf(mPhi)*sinf(mTheta);
157 float y = mRadius*cosf(mPhi);
158
159 // Build the view matrix.
160 XMVECTOR pos = XMVectorSet(x, y, z, 1.0f);
161 XMVECTOR target = XMVectorZero();
162 XMVECTOR up = XMVectorSet(0.0f, 1.0f, 0.0f, 0.0f);
163
164 XMMATRIX view = XMMatrixLookAtLH(pos, target, up);
165 XMStoreFloat4x4(&mView, view);
166
167
168
169
170 XMMATRIX world = XMLoadFloat4x4(&mWorld);
171 XMMATRIX proj = XMLoadFloat4x4(&mProj);
172 XMMATRIX worldViewProj = world*view*proj;
173
174 // Update the constant buffer with the latest worldViewProj matrix.
175 ObjectConstants objConstants;
176 XMStoreFloat4x4(&objConstants.WorldViewProj, XMMatrixTranspose(worldViewProj));
177 mObjectCB->CopyData(0, objConstants);
178
179}

181void BoxApp::Draw(const GameTimer& gt)
182{
183 // Reuse the memory associated with command recording.
184 // We can only reset when the associated command lists have finished execution on the GPU.
185 ThrowIfFailed(mDirectCmdListAlloc->Reset());
186
187 //step3: A command list can be reset after it has been added to the command queue via ExecuteCommandList.
188 // Reusing the command list reuses memory.
189 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), mPSO.Get()));
190
191 mCommandList->RSSetViewports(1, &mScreenViewport);
192 mCommandList->RSSetScissorRects(1, &mScissorRect);
193
194 // Indicate a state transition on the resource usage.
195 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
196 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
197
198
199 // Clear the back buffer and depth buffer.
200 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), Colors::LightSteelBlue, 0, nullptr);
201 mCommandList->ClearDepthStencilView(DepthStencilView(), D3D12_CLEAR_FLAG_DEPTH | D3D12_CLEAR_FLAG_STENCIL, 1.0f, 0, 0, nullptr);
202
203 // Specify the buffers we are going to render to.
204 mCommandList->OMSetRenderTargets(1, &CurrentBackBufferView(), true, &DepthStencilView());
205
206 ID3D12DescriptorHeap* descriptorHeaps[] = { mCbvHeap.Get() };
207 mCommandList->SetDescriptorHeaps(_countof(descriptorHeaps), descriptorHeaps);
208
209 mCommandList->SetGraphicsRootSignature(mRootSignature.Get());
210
211 mCommandList->IASetVertexBuffers(0, 1, &mBoxGeo->VertexBufferView());
212 mCommandList->IASetIndexBuffer(&mBoxGeo->IndexBufferView());
213 mCommandList->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
214
215 mCommandList->SetGraphicsRootDescriptorTable(0, mCbvHeap->GetGPUDescriptorHandleForHeapStart());
216
217 mCommandList->DrawIndexedInstanced(
218 mBoxGeo->DrawArgs["box"].IndexCount,
219 1, 0, 0, 0);
220
221 // Indicate a state transition on the resource usage.
222 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
223 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
224
225 //step4: Done recording commands. Comment this line out to see what debugger will tell you in the output window.
226 ThrowIfFailed(mCommandList->Close());
227
228 // Add the command list to the queue for execution.
229 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
230 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
231
232 // swap the back and front buffers
233 //HRESULT Present(UINT SyncInterval,UINT Flags);
234 //SyncInterval: 0 - The presentation occurs immediately, there is no synchronization.
235 //Flags defined by the DXGI_PRESENT constants. 0: Present a frame from each buffer (starting with the current buffer) to the output.
236 ThrowIfFailed(mSwapChain->Present(0, 0));
237 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
238
239 // Wait until frame commands are complete. This waiting is inefficient and is
240 // done for simplicity. Later we will show how to organize our rendering code
241 // so we do not have to wait per frame.
242 FlushCommandQueue();
243}

260void BoxApp::OnMouseMove(WPARAM btnState, int x, int y)
261{
262 if((btnState & MK_LBUTTON) != 0)
263 {
264 // Make each pixel correspond to a quarter of a degree.
265 float dx = XMConvertToRadians(0.25f*static_cast<float>(x - mLastMousePos.x));
266 float dy = XMConvertToRadians(0.25f*static_cast<float>(y - mLastMousePos.y));
267
268 // Update angles based on input to orbit camera around box.
269 mTheta += dx;
270 mPhi += dy;
271
272 // Restrict the angle mPhi.
273 mPhi = MathHelper::Clamp(mPhi, 0.1f, MathHelper::Pi - 0.1f);
274 }
275 else if((btnState & MK_RBUTTON) != 0)
276 {
277 // Make each pixel correspond to 0.005 unit in the scene.
278 float dx = 0.005f*static_cast<float>(x - mLastMousePos.x);
279 float dy = 0.005f*static_cast<float>(y - mLastMousePos.y);
280
281 // Update the camera radius based on input.
282 mRadius += dx - dy;
283
284 // Restrict the radius.
285 mRadius = MathHelper::Clamp(mRadius, 3.0f, 15.0f);
286 }
287
288 mLastMousePos.x = x;
289 mLastMousePos.y = y;
290}

325void BoxApp::BuildRootSignature()
326{
327 // Shader programs typically require resources as input (constant buffers,
328 // textures, samplers). The root signature defines the resources the shader
329 // programs expect. If we think of the shader programs as a function, and
330 // the input resources as function parameters, then the root signature can be
331 // thought of as defining the function signature.
332
333 // Root parameter can be a table, root descriptor or root constants.
334 CD3DX12_ROOT_PARAMETER slotRootParameter[1];
335
336 // Create a single descriptor table of CBVs.
337 CD3DX12_DESCRIPTOR_RANGE cbvTable;
338 cbvTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_CBV, 1, 0);
339 slotRootParameter[0].InitAsDescriptorTable(1, &cbvTable);
340
341 // A root signature is an array of root parameters.
342 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(1, slotRootParameter, 0, nullptr,
343 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
344
345
346 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
347 ComPtr<ID3DBlob> serializedRootSig = nullptr;
348 ComPtr<ID3DBlob> errorBlob = nullptr;
349 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
350 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
351
352 if(errorBlob != nullptr)
353 {
354 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
355 }
356 ThrowIfFailed(hr);
357
358 ThrowIfFailed(md3dDevice->CreateRootSignature(
359 0,
360 serializedRootSig->GetBufferPointer(),
361 serializedRootSig->GetBufferSize(),
362 IID_PPV_ARGS(&mRootSignature)));
363}

392{
393 std::array<Vertex, 8> vertices =
394 {
395 Vertex({ XMFLOAT3(-1.0f, -1.0f, -1.0f), XMFLOAT4(Colors::White) }),
396 Vertex({ XMFLOAT3(-1.0f, +1.0f, -1.0f), XMFLOAT4(Colors::Black) }),

401 Vertex({ XMFLOAT3(+1.0f, +1.0f, +1.0f), XMFLOAT4(Colors::Cyan) }),

404
405 std::array<std::uint16_t, 36> indices =
406 {
407 // front face
408 0, 1, 2,
409 0, 2, 3,
410
411 // back face
412 4, 6, 5,
413 4, 7, 6,
414
415 // left face
416 4, 5, 1,
417 4, 1, 0,
418
419 // right face
420 3, 2, 6,
421 3, 6, 7,
422
423 // top face
424 1, 5, 6,
425 1, 6, 2,
426
427 // bottom face
428 4, 0, 3,
429 4, 3, 7
430 };
431
432 const UINT vbByteSize = (UINT)vertices.size() * sizeof(Vertex);
433 const UINT ibByteSize = (UINT)indices.size() * sizeof(std::uint16_t);
434
435 mBoxGeo = std::make_unique<MeshGeometry>();
436 mBoxGeo->Name = "boxGeo";
437

449
450 mBoxGeo->VertexByteStride = sizeof(Vertex);
