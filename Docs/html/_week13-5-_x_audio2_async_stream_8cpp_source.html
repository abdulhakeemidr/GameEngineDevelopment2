82// Entry point to the program
83//--------------------------------------------------------------------------------------
84int main()
85{
86 //
87 // Initialize XAudio2
88 //
89 HRESULT hr = CoInitializeEx(nullptr, COINIT_MULTITHREADED);
90 if (FAILED(hr))
91 {
92 wprintf(L"Failed to init COM: %#X\n", hr);
93 return 0;
94 }
95
96#if defined(USING_XAUDIO2_7_DIRECTX)
97 // Workaround for XAudio 2.7 known issue
98#ifdef _DEBUG
99 HMODULE mXAudioDLL = LoadLibraryExW(L"XAudioD2_7.DLL", nullptr, 0x00000800 /* LOAD_LIBRARY_SEARCH_SYSTEM32 */);
100#else
101 HMODULE mXAudioDLL = LoadLibraryExW(L"XAudio2_7.DLL", nullptr, 0x00000800 /* LOAD_LIBRARY_SEARCH_SYSTEM32 */);
102#endif
103 if (!mXAudioDLL)
104 {
105 wprintf(L"Failed to find XAudio 2.7 DLL");
106 CoUninitialize();
107 return 0;
108 }
109#endif
110
111 UINT32 flags = 0;
112#if defined(USING_XAUDIO2_7_DIRECTX) && defined(_DEBUG)
113 flags |= XAUDIO2_DEBUG_ENGINE;
114#endif
115
116 ComPtr<IXAudio2> pXAudio2;
117 hr = XAudio2Create(pXAudio2.GetAddressOf(), flags);
118 if (FAILED(hr))
119 {
120 wprintf(L"Failed to init XAudio2 engine: %#X\n", hr);
121 CoUninitialize();
122 return 0;
123 }
124
125#if !defined(USING_XAUDIO2_7_DIRECTX) && defined(_DEBUG)
126 // To see the trace output, you need to view ETW logs for this application:
127 // Go to Control Panel, Administrative Tools, Event Viewer.
128 // View->Show Analytic and Debug Logs.
129 // Applications and Services Logs / Microsoft / Windows / XAudio2.
130 // Right click on Microsoft Windows XAudio2 debug logging, Properties, then Enable Logging, and hit OK
131 XAUDIO2_DEBUG_CONFIGURATION debug = { 0 };
132 debug.TraceMask = XAUDIO2_LOG_ERRORS | XAUDIO2_LOG_WARNINGS;
133 debug.BreakMask = XAUDIO2_LOG_ERRORS;
134 pXAudio2->SetDebugConfiguration(&debug, 0);
135#endif
136
137 //
138 // Create a mastering voice
139 //
140 IXAudio2MasteringVoice* pMasteringVoice = nullptr;
141
142 if (FAILED(hr = pXAudio2->CreateMasteringVoice(&pMasteringVoice)))
143 {
144 wprintf(L"Failed creating mastering voice: %#X\n", hr);
145 pXAudio2.Reset();
146 CoUninitialize();
147 return 0;
148 }
149
150 //
151 // Find our wave bank file
152 //
153 WCHAR wavebank[MAX_PATH];
154
155 if (FAILED(hr = FindMediaFileCch(wavebank, MAX_PATH, L"Media\\Banks\\wavebank.xwb")))
156 {
157 wprintf(L"Failed to find media file (%#X)\n", hr);
158 pXAudio2.Reset();
159 CoUninitialize();
160 return 0;
161 }
162
163 //
164 // Extract wavebank data (entries, formats, offsets, and sizes)
165 //
166 // Note we are using Wave Banks to get sector-aligned streaming data to allow us to use
167 // async unbuffered I/O. Raw .WAV files do not meet these requirements.
168 //
169 WaveBankReader wb;
170
171 if (FAILED(hr = wb.Open(wavebank)))
172 {
173 wprintf(L"Failed to wavebank data (%#X)\n", hr);
174 pXAudio2.Reset();
175 CoUninitialize();
176 return 0;
177 }
178
179 wprintf(L"Wavebank loaded with %u entries.\n", wb.Count());
180
181 if (!wb.IsStreamingBank())
182 {
183 wprintf(L"This sample plays back streaming wave banks.\nSee XAudio2WaveBank for playing in-memory wave banks");
184 pXAudio2.Reset();
185 CoUninitialize();
186 return 0;
187 }
188
189 wprintf(L"Press <ESC> to exit.\n");
190
191 //
192 // Repeated loop through all the wavebank entries
193 //
194 bool exit = false;
195
196 while (!exit)
197 {
198 for (DWORD i = 0; i < wb.Count(); ++i)
199 {
200 wprintf(L"Now playing wave entry %u", i);
201
202 //
203 // Get the info we need to play back this wave (need enough space for PCM, ADPCM, and xWMA formats)
204 //
205 char formatBuff[64];
206 WAVEFORMATEX* wfx = reinterpret_cast<WAVEFORMATEX*>(&formatBuff);
207
208 if (FAILED(hr = wb.GetFormat(i, wfx, 64)))
209 {
210 wprintf(L"\nCouldn't get wave format for entry %u: error 0x%x\n", i, hr);
211 exit = true;
212 break;
213 }
214
215 WaveBankReader::Metadata metadata;
216 if (FAILED(hr = wb.GetMetadata(i, metadata)))
217 {
218 wprintf(L"\nCouldn't get metadta for entry %u: error 0x%x\n", i, hr);
219 exit = true;
220 break;
221 }
222
223 //
224 // Create an XAudio2 voice to stream this wave
225 //
226 StreamingVoiceContext voiceContext;
227
228 IXAudio2SourceVoice* pSourceVoice;
229 if (FAILED(hr = pXAudio2->CreateSourceVoice(&pSourceVoice, wfx, 0, 1.0f, &voiceContext)))
230 {
231 wprintf(L"\nError %#X creating source voice\n", hr);
232 exit = true;
233 break;
234 }
235 pSourceVoice->Start(0, 0);
236
237 //
238 // Create an overlapped structure and buffers to handle the async I/O
239 //
240 OVERLAPPED ovlCurrentRequest = { 0 };
241#if (_WIN32_WINNT >= _WIN32_WINNT_VISTA)
242 ovlCurrentRequest.hEvent = CreateEventEx(nullptr, nullptr, CREATE_EVENT_MANUAL_RESET, EVENT_MODIFY_STATE | SYNCHRONIZE);
243#else
244 ovlCurrentRequest.hEvent = CreateEvent(nullptr, TRUE, FALSE, nullptr);
245#endif
246
247 if ((STREAMING_BUFFER_SIZE % wfx->nBlockAlign) != 0)
248 {
249 //
250 // non-PCM data will fail here. ADPCM requires a more complicated streaming mechanism to deal with submission in audio frames that do
251 // not necessarily align to the 2K async boundary.
252 //
253 wprintf(L"\nStreaming buffer size (%u) is not aligned with sample block requirements (%u)\n", STREAMING_BUFFER_SIZE, wfx->nBlockAlign);
254 exit = true;
255 break;
256 }
257
258 std::unique_ptr<uint8_t[]> buffers[MAX_BUFFER_COUNT];
259 for (size_t j = 0; j < MAX_BUFFER_COUNT; ++j)
260 {
261 buffers[j].reset(new uint8_t[STREAMING_BUFFER_SIZE]);
262 }
263 DWORD currentDiskReadBuffer = 0;
264 DWORD currentPosition = 0;
265
266 //
267 // This sample code shows the simplest way to manage asynchronous
268 // streaming. There are three different processes involved. One is the management
269 // process, which is what we're writing here. The other two processes are
270 // essentially hardware operations: disk reads from the I/O system, and
271 // audio processing from XAudio2. Disk reads and audio playback both happen
272 // without much intervention from our application, so our job is just to make
273 // sure that the data being read off the disk makes it over to the audio
274 // processor in time to be played back.
275 //
276 // There are two events that can happen in this system. The disk I/O system can
277 // signal that data is ready, and the audio system can signal that it's done
278 // playing back data. We can handle either or both of these events either synchronously
279 // (via polling) or asynchronously (via callbacks or by waiting on an event
280 // object).
281 //
282 HANDLE async = wb.GetAsyncHandle();
283
284 while (currentPosition < metadata.lengthBytes)
285 {
286 wprintf(L".");
287
288 if (GetAsyncKeyState(VK_ESCAPE))
289 {
290 exit = true;
291
292 while (GetAsyncKeyState(VK_ESCAPE))
293 Sleep(10);
294
295 break;
296 }
297
298 //
299 // Issue a request.
300 //
301 // Note: although the file read will be done asynchronously, it is possible for the
302 // call to ReadFile to block for longer than you might think. If the I/O system needs
303 // to read the file allocation table in order to satisfy the read, it will do that
304 // BEFORE returning from ReadFile. That means that this call could potentially
305 // block for several milliseconds! In order to get "true" async I/O you should put
306 // this entire loop on a separate thread.
307 //
308 // Second note: async requests have to be a multiple of the disk sector size. Rather than
309 // handle this conditionally, make all reads the same size but remember how many
310 // bytes we actually want and only submit that many to the voice.
311 //
312 DWORD cbValid = min(STREAMING_BUFFER_SIZE, metadata.lengthBytes - currentPosition);
313 ovlCurrentRequest.Offset = metadata.offsetBytes + currentPosition;
314
315 bool wait = false;
316 if (!ReadFile(async, buffers[currentDiskReadBuffer].get(), STREAMING_BUFFER_SIZE, nullptr, &ovlCurrentRequest))
317 {
318 DWORD error = GetLastError();
319 if (error != ERROR_IO_PENDING)
320 {
321 wprintf(L"\nCouldn't start async read: error %#X\n", HRESULT_FROM_WIN32(error));
322 exit = true;
323 break;
324 }
325 wait = true;
326 }
327
328 currentPosition += cbValid;
329
330 //
331 // At this point the read is progressing in the background and we are free to do
332 // other processing while we wait for it to finish. For the purposes of this sample,
333 // however, we'll just go to sleep until the read is done.
334 //
335 if (wait)
336 WaitForSingleObject(ovlCurrentRequest.hEvent, INFINITE);
337
338 DWORD cb;
339#if (_WIN32_WINNT >= 0x0602 /*_WIN32_WINNT_WIN8*/)
340 BOOL result = GetOverlappedResultEx(async, &ovlCurrentRequest, &cb, 0, FALSE);
341#else
342 BOOL result = GetOverlappedResult(async, &ovlCurrentRequest, &cb, FALSE);
343#endif
344 if (!result)
345 {
346 wprintf(L"\nFailed waiting for async read: error %#X\n", HRESULT_FROM_WIN32(GetLastError()));
347 exit = true;
348 break;
349 }
350
351 //
352 // Now that the event has been signaled, we know we have audio available. The next
353 // question is whether our XAudio2 source voice has played enough data for us to give
354 // it another buffer full of audio. We'd like to keep no more than MAX_BUFFER_COUNT - 1
355 // buffers on the queue, so that one buffer is always free for disk I/O.
356 //
357 XAUDIO2_VOICE_STATE state;
358 for (;; )
359 {
360 pSourceVoice->GetState(&state);
361 if (state.BuffersQueued < MAX_BUFFER_COUNT - 1)
362 break;
363
364 WaitForSingleObject(voiceContext.hBufferEndEvent, INFINITE);
365 }
366
367 //
368 // At this point we have a buffer full of audio and enough room to submit it, so
369 // let's submit it and get another read request going.
370 //
371 XAUDIO2_BUFFER buf = { 0 };
372 buf.AudioBytes = cbValid;
373 buf.pAudioData = buffers[currentDiskReadBuffer].get();
374 if (currentPosition >= metadata.lengthBytes)
375 buf.Flags = XAUDIO2_END_OF_STREAM;
376
377 pSourceVoice->SubmitSourceBuffer(&buf);
378
379 currentDiskReadBuffer++;
380 currentDiskReadBuffer %= MAX_BUFFER_COUNT;
381 }
382
383 if (!exit)
384 {
385 wprintf(L"done streaming..");
386
387 XAUDIO2_VOICE_STATE state;
388 for (; ; )
389 {
390 pSourceVoice->GetState(&state);
391 if (!state.BuffersQueued)
392 break;
393
394 wprintf(L".");
395 WaitForSingleObject(voiceContext.hBufferEndEvent, INFINITE);
396 }
397 }
398
399 //
400 // Clean up
401 //
402 pSourceVoice->Stop(0);
403 pSourceVoice->DestroyVoice();
404
405 CloseHandle(ovlCurrentRequest.hEvent);
406
407 wprintf(L"stopped\n");
408
409 if (exit)
410 break;
411
412 Sleep(500);
413 }
414 }
415
416 //
417 // Cleanup XAudio2
418 //
419
420 // All XAudio2 interfaces are released when the engine is destroyed, but being tidy
421 pMasteringVoice->DestroyVoice();
422
423 pXAudio2.Reset();
424
425#ifdef USING_XAUDIO2_7_DIRECTX
426 if (mXAudioDLL)
427 FreeLibrary(mXAudioDLL);
428#endif
429
430 CoUninitialize();
431}

437_Use_decl_annotations_
438HRESULT FindMediaFileCch(WCHAR* strDestPath, int cchDest, LPCWSTR strFilename)
439{
440 bool bFound = false;
441
442 if (!strFilename || strFilename[0] == 0 || !strDestPath || cchDest < 10)
443 return E_INVALIDARG;
444
445 // Get the exe name, and exe path
446 WCHAR strExePath[MAX_PATH] = { 0 };
447 WCHAR strExeName[MAX_PATH] = { 0 };
448 WCHAR* strLastSlash = nullptr;
449 GetModuleFileName(nullptr, strExePath, MAX_PATH);
450 strExePath[MAX_PATH - 1] = 0;
451 strLastSlash = wcsrchr(strExePath, TEXT('\\'));
452 if (strLastSlash)
453 {
454 wcscpy_s(strExeName, MAX_PATH, &strLastSlash[1]);
455
456 // Chop the exe name from the exe path
457 *strLastSlash = 0;
458
459 // Chop the .exe from the exe name
460 strLastSlash = wcsrchr(strExeName, TEXT('.'));
461 if (strLastSlash)
462 *strLastSlash = 0;
463 }
464
465 wcscpy_s(strDestPath, cchDest, strFilename);
466 if (GetFileAttributes(strDestPath) != 0xFFFFFFFF)
467 return S_OK;
468
469 // Search all parent directories starting at .\ and using strFilename as the leaf name
470 WCHAR strLeafName[MAX_PATH] = { 0 };
471 wcscpy_s(strLeafName, MAX_PATH, strFilename);
472
473 WCHAR strFullPath[MAX_PATH] = { 0 };
474 WCHAR strFullFileName[MAX_PATH] = { 0 };
475 WCHAR strSearch[MAX_PATH] = { 0 };
476 WCHAR* strFilePart = nullptr;
477
478 GetFullPathName(L".", MAX_PATH, strFullPath, &strFilePart);
479 if (!strFilePart)
480 return E_FAIL;
481
482 while (strFilePart && *strFilePart != '\0')
483 {
484 swprintf_s(strFullFileName, MAX_PATH, L"%s\\%s", strFullPath, strLeafName);
485 if (GetFileAttributes(strFullFileName) != 0xFFFFFFFF)
486 {
487 wcscpy_s(strDestPath, cchDest, strFullFileName);
488 bFound = true;
489 break;
490 }
491
492 swprintf_s(strFullFileName, MAX_PATH, L"%s\\%s\\%s", strFullPath, strExeName, strLeafName);
493 if (GetFileAttributes(strFullFileName) != 0xFFFFFFFF)
494 {
495 wcscpy_s(strDestPath, cchDest, strFullFileName);
496 bFound = true;
497 break;
498 }
499
500 swprintf_s(strSearch, MAX_PATH, L"%s\\..", strFullPath);
501 GetFullPathName(strSearch, MAX_PATH, strFullPath, &strFilePart);
502 }
503 if (bFound)
504 return S_OK;
505
506 // On failure, return the file as the path but also return an error code
507 wcscpy_s(strDestPath, cchDest, strFilename);
508
509 return HRESULT_FROM_WIN32(ERROR_FILE_NOT_FOUND);
510}
