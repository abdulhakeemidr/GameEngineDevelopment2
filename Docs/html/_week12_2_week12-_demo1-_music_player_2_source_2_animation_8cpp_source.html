102 mElapsedTime += dt;
103

106

109
110 // While we have a frame to process

119 // Move it down one line
120 textureRect.left = 0;

122 }
123
124 // And progress to next frame
125 mElapsedTime -= timePerFrame;
126 if (mRepeat)
