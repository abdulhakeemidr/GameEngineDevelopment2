120 const RIFFChunk* FindChunk(
121 _In_reads_bytes_(sizeBytes) const uint8_t* data,
122 _In_ size_t sizeBytes,
123 _In_ uint32_t tag) noexcept
124 {
125 if (!data)
126 return nullptr;
127
128 const uint8_t* ptr = data;
129 const uint8_t* end = data + sizeBytes;
130
131 while (end > (ptr + sizeof(RIFFChunk)))
132 {
133 auto header = reinterpret_cast<const RIFFChunk*>(ptr);
134 if (header->tag == tag)
135 return header;
136
137 auto offset = header->size + sizeof(RIFFChunk);
138 ptr += offset;
139 }
140
141 return nullptr;
142 }

146 HRESULT WaveFindFormatAndData(
147 _In_reads_bytes_(wavDataSize) const uint8_t* wavData,
148 _In_ size_t wavDataSize,
149 _Outptr_ const WAVEFORMATEX** pwfx,
150 _Outptr_ const uint8_t** pdata,
151 _Out_ uint32_t* dataSize,
152 _Out_ bool& dpds,
153 _Out_ bool& seek) noexcept
154 {
155 if (!wavData || !pwfx)
156 return E_POINTER;
157
158 dpds = seek = false;
159
160 if (wavDataSize < (sizeof(RIFFChunk) * 2 + sizeof(uint32_t) + sizeof(WAVEFORMAT)))
161 {
162 return E_FAIL;
163 }
164
165 const uint8_t* wavEnd = wavData + wavDataSize;
166
167 // Locate RIFF 'WAVE'
168 auto riffChunk = FindChunk(wavData, wavDataSize, FOURCC_RIFF_TAG);
169 if (!riffChunk || riffChunk->size < 4)
170 {
171 return E_FAIL;
172 }
173
174 auto riffHeader = reinterpret_cast<const RIFFChunkHeader*>(riffChunk);
175 if (riffHeader->riff != FOURCC_WAVE_FILE_TAG && riffHeader->riff != FOURCC_XWMA_FILE_TAG)
176 {
177 return E_FAIL;
178 }
179
180 // Locate 'fmt '
181 auto ptr = reinterpret_cast<const uint8_t*>(riffHeader) + sizeof(RIFFChunkHeader);
182 if ((ptr + sizeof(RIFFChunk)) > wavEnd)
183 {
184 return HRESULT_FROM_WIN32(ERROR_HANDLE_EOF);
185 }
186
187 auto fmtChunk = FindChunk(ptr, riffHeader->size, FOURCC_FORMAT_TAG);
188 if (!fmtChunk || fmtChunk->size < sizeof(PCMWAVEFORMAT))
189 {
190 return E_FAIL;
191 }
192
193 ptr = reinterpret_cast<const uint8_t*>(fmtChunk) + sizeof(RIFFChunk);
194 if (ptr + fmtChunk->size > wavEnd)
195 {
196 return HRESULT_FROM_WIN32(ERROR_HANDLE_EOF);
197 }
198
199 auto wf = reinterpret_cast<const WAVEFORMAT*>(ptr);
200
201 // Validate WAVEFORMAT (focused on chunk size and format tag, not other data that XAUDIO2 will validate)
202 switch (wf->wFormatTag)
203 {
204 case WAVE_FORMAT_PCM:
205 case WAVE_FORMAT_IEEE_FLOAT:
206 // Can be a PCMWAVEFORMAT (16 bytes) or WAVEFORMATEX (18 bytes)
207 // We validiated chunk as at least sizeof(PCMWAVEFORMAT) above
208 break;
209
210 default:
211 {
212 if (fmtChunk->size < sizeof(WAVEFORMATEX))
213 {
214 return E_FAIL;
215 }
216
217 auto wfx = reinterpret_cast<const WAVEFORMATEX*>(ptr);
218
219 if (fmtChunk->size < (sizeof(WAVEFORMATEX) + wfx->cbSize))
220 {
221 return E_FAIL;
222 }
223
224 switch (wfx->wFormatTag)
225 {
226 case WAVE_FORMAT_WMAUDIO2:
227 case WAVE_FORMAT_WMAUDIO3:
228 dpds = true;
229 break;
230
231 case 0x166 /*WAVE_FORMAT_XMA2*/: // XMA2 is supported by Xbox One
232 if ((fmtChunk->size < 52 /*sizeof(XMA2WAVEFORMATEX)*/) || (wfx->cbSize < 34 /*( sizeof(XMA2WAVEFORMATEX) - sizeof(WAVEFORMATEX) )*/))
233 {
234 return E_FAIL;
235 }
236 seek = true;
237 break;
238
239 case WAVE_FORMAT_ADPCM:
240 if ((fmtChunk->size < (sizeof(WAVEFORMATEX) + 32)) || (wfx->cbSize < 32 /*MSADPCM_FORMAT_EXTRA_BYTES*/))
241 {
242 return E_FAIL;
243 }
244 break;
245
246 case WAVE_FORMAT_EXTENSIBLE:
247 if ((fmtChunk->size < sizeof(WAVEFORMATEXTENSIBLE)) || (wfx->cbSize < (sizeof(WAVEFORMATEXTENSIBLE) - sizeof(WAVEFORMATEX))))
248 {
249 return E_FAIL;
250 }
251 else
252 {
253 static const GUID s_wfexBase = { 0x00000000, 0x0000, 0x0010, { 0x80, 0x00, 0x00, 0xAA, 0x00, 0x38, 0x9B, 0x71 } };
254
255 auto wfex = reinterpret_cast<const WAVEFORMATEXTENSIBLE*>(ptr);
256
257 if (memcmp(reinterpret_cast<const BYTE*>(&wfex->SubFormat) + sizeof(DWORD),
258 reinterpret_cast<const BYTE*>(&s_wfexBase) + sizeof(DWORD), sizeof(GUID) - sizeof(DWORD)) != 0)
259 {
260 return HRESULT_FROM_WIN32(ERROR_NOT_SUPPORTED);
261 }
262
263 switch (wfex->SubFormat.Data1)
264 {
265 case WAVE_FORMAT_PCM:
266 case WAVE_FORMAT_IEEE_FLOAT:
267 break;
268
269 // MS-ADPCM and XMA2 are not supported as WAVEFORMATEXTENSIBLE
270
271 case WAVE_FORMAT_WMAUDIO2:
272 case WAVE_FORMAT_WMAUDIO3:
273 dpds = true;
274 break;
275
276 default:
277 return HRESULT_FROM_WIN32(ERROR_NOT_SUPPORTED);
278 }
279
280 }
281 break;
282
283 default:
284 return HRESULT_FROM_WIN32(ERROR_NOT_SUPPORTED);
285 }
286 }
287 }
288
289 // Locate 'data'
290 ptr = reinterpret_cast<const uint8_t*>(riffHeader) + sizeof(RIFFChunkHeader);
291 if ((ptr + sizeof(RIFFChunk)) > wavEnd)
292 {
293 return HRESULT_FROM_WIN32(ERROR_HANDLE_EOF);
294 }
295
296 auto dataChunk = FindChunk(ptr, riffChunk->size, FOURCC_DATA_TAG);
297 if (!dataChunk || !dataChunk->size)
298 {
299 return HRESULT_FROM_WIN32(ERROR_INVALID_DATA);
300 }
301
302 ptr = reinterpret_cast<const uint8_t*>(dataChunk) + sizeof(RIFFChunk);
303 if (ptr + dataChunk->size > wavEnd)
304 {
305 return HRESULT_FROM_WIN32(ERROR_HANDLE_EOF);
306 }
307
308 *pwfx = reinterpret_cast<const WAVEFORMATEX*>(wf);
309 *pdata = ptr;
310 *dataSize = dataChunk->size;
311 return S_OK;
312 }

316 HRESULT WaveFindLoopInfo(
317 _In_reads_bytes_(wavDataSize) const uint8_t* wavData,
318 _In_ size_t wavDataSize,
319 _Out_ uint32_t* pLoopStart,
320 _Out_ uint32_t* pLoopLength) noexcept
321 {
322 if (!wavData || !pLoopStart || !pLoopLength)
323 return E_POINTER;
324
325 if (wavDataSize < (sizeof(RIFFChunk) + sizeof(uint32_t)))
326 {
327 return E_FAIL;
328 }
329
330 *pLoopStart = 0;
331 *pLoopLength = 0;
332
333 const uint8_t* wavEnd = wavData + wavDataSize;
334
335 // Locate RIFF 'WAVE'
336 auto riffChunk = FindChunk(wavData, wavDataSize, FOURCC_RIFF_TAG);
337 if (!riffChunk || riffChunk->size < 4)
338 {
339 return E_FAIL;
340 }
341
342 auto riffHeader = reinterpret_cast<const RIFFChunkHeader*>(riffChunk);
343 if (riffHeader->riff == FOURCC_XWMA_FILE_TAG)
344 {
345 // xWMA files do not contain loop information
346 return S_OK;
347 }
348
349 if (riffHeader->riff != FOURCC_WAVE_FILE_TAG)
350 {
351 return E_FAIL;
352 }
353
354 // Locate 'wsmp' (DLS Chunk)
355 auto ptr = reinterpret_cast<const uint8_t*>(riffHeader) + sizeof(RIFFChunkHeader);
356 if ((ptr + sizeof(RIFFChunk)) > wavEnd)
357 {
358 return HRESULT_FROM_WIN32(ERROR_HANDLE_EOF);
359 }
360
361 auto dlsChunk = FindChunk(ptr, riffChunk->size, FOURCC_DLS_SAMPLE);
362 if (dlsChunk)
363 {
364 ptr = reinterpret_cast<const uint8_t*>(dlsChunk) + sizeof(RIFFChunk);
365 if (ptr + dlsChunk->size > wavEnd)
366 {
367 return HRESULT_FROM_WIN32(ERROR_HANDLE_EOF);
368 }
369
370 if (dlsChunk->size >= sizeof(RIFFDLSSample))
371 {
372 auto dlsSample = reinterpret_cast<const RIFFDLSSample*>(ptr);
373
374 if (dlsChunk->size >= (dlsSample->size + dlsSample->loopCount * sizeof(DLSLoop)))
375 {
376 auto loops = reinterpret_cast<const DLSLoop*>(ptr + dlsSample->size);
377 for (uint32_t j = 0; j < dlsSample->loopCount; ++j)
378 {
379 if ((loops[j].loopType == DLSLoop::LOOP_TYPE_FORWARD || loops[j].loopType == DLSLoop::LOOP_TYPE_RELEASE))
380 {
381 // Return 'forward' loop
382 *pLoopStart = loops[j].loopStart;
383 *pLoopLength = loops[j].loopLength;
384 return S_OK;
385 }
386 }
387 }
388 }
389 }
390
391 // Locate 'smpl' (Sample Chunk)
392 auto midiChunk = FindChunk(ptr, riffChunk->size, FOURCC_MIDI_SAMPLE);
393 if (midiChunk)
394 {
395 ptr = reinterpret_cast<const uint8_t*>(midiChunk) + sizeof(RIFFChunk);
396 if (ptr + midiChunk->size > wavEnd)
397 {
398 return HRESULT_FROM_WIN32(ERROR_HANDLE_EOF);
399 }
400
401 if (midiChunk->size >= sizeof(RIFFMIDISample))
402 {
403 auto midiSample = reinterpret_cast<const RIFFMIDISample*>(ptr);
404
405 if (midiChunk->size >= (sizeof(RIFFMIDISample) + midiSample->loopCount * sizeof(MIDILoop)))
406 {
407 auto loops = reinterpret_cast<const MIDILoop*>(ptr + sizeof(RIFFMIDISample));
408 for (uint32_t j = 0; j < midiSample->loopCount; ++j)
409 {
410 if (loops[j].type == MIDILoop::LOOP_TYPE_FORWARD)
411 {
412 // Return 'forward' loop
413 *pLoopStart = loops[j].start;
414 *pLoopLength = loops[j].end - loops[j].start + 1;
415 return S_OK;
416 }
417 }
418 }
419 }
420 }
421
422 return S_OK;
423 }

490 HRESULT LoadAudioFromFile(
491 _In_z_ const wchar_t* szFileName,
492 _Inout_ std::unique_ptr<uint8_t[]>& wavData,
493 _Out_ DWORD* bytesRead) noexcept
494 {
495 if (!szFileName)
496 return E_INVALIDARG;
497
498 // open the file
499 #if (_WIN32_WINNT >= _WIN32_WINNT_WIN8)
500 ScopedHandle hFile(safe_handle(CreateFile2(szFileName,
501 GENERIC_READ,
502 FILE_SHARE_READ,
503 OPEN_EXISTING,
504 nullptr)));
505 #else
506 ScopedHandle hFile(safe_handle(CreateFileW(szFileName,
507 GENERIC_READ,
508 FILE_SHARE_READ,
509 nullptr,
510 OPEN_EXISTING,
511 FILE_ATTRIBUTE_NORMAL,
512 nullptr)));
513 #endif
514
515 if (!hFile)
516 {
517 return HRESULT_FROM_WIN32(GetLastError());
518 }
519
520 // Get the file size
521 FILE_STANDARD_INFO fileInfo;
522 if (!GetFileInformationByHandleEx(hFile.get(), FileStandardInfo, &fileInfo, sizeof(fileInfo)))
523 {
524 return HRESULT_FROM_WIN32(GetLastError());
525 }
526
527 // File is too big for 32-bit allocation, so reject read
528 if (fileInfo.EndOfFile.HighPart > 0)
529 {
530 return E_FAIL;
531 }
532
533 // Need at least enough data to have a valid minimal WAV file
534 if (fileInfo.EndOfFile.LowPart < (sizeof(RIFFChunk) * 2 + sizeof(DWORD) + sizeof(WAVEFORMAT)))
535 {
536 return E_FAIL;
537 }
538
539 // create enough space for the file data
540 wavData.reset(new (std::nothrow) uint8_t[fileInfo.EndOfFile.LowPart]);
541 if (!wavData)
542 {
543 return E_OUTOFMEMORY;
544 }
545
546 // read the data in
547 if (!ReadFile(hFile.get(),
548 wavData.get(),
549 fileInfo.EndOfFile.LowPart,
550 bytesRead,
551 nullptr
552 ))
553 {
554 return HRESULT_FROM_WIN32(GetLastError());
555 }
556
557 return (*bytesRead < fileInfo.EndOfFile.LowPart) ? E_FAIL : S_OK;
558 }
559}
