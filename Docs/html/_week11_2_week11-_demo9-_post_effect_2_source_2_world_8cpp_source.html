143bool matchesCategories(SceneNode::Pair& colliders, Category::Type type1, Category::Type type2)
144{
145 unsigned int category1 = colliders.first->getCategory();
146 unsigned int category2 = colliders.second->getCategory();
147
148 // Make sure first pair entry has category type1 and second has type2
149 if (type1 & category1 && type2 & category2)
150 {
151 return true;
152 }
153 else if (type1 & category2 && type2 & category1)
154 {
155 std::swap(colliders.first, colliders.second);
156 return true;
157 }
158 else
159 {
160 return false;
161 }

164void World::handleCollisions()
165{
166 std::set<SceneNode::Pair> collisionPairs;
167 mSceneGraph.checkSceneCollision(mSceneGraph, collisionPairs);
168
169 for(SceneNode::Pair pair: collisionPairs)
170 {
171 if (matchesCategories(pair, Category::PlayerAircraft, Category::EnemyAircraft))
172 {
173 auto& player = static_cast<Aircraft&>(*pair.first);
174 auto& enemy = static_cast<Aircraft&>(*pair.second);
175
176 // Collision: Player damage = enemy's remaining HP
177 player.damage(enemy.getHitpoints());
178 enemy.destroy();
179 }
180
181 else if (matchesCategories(pair, Category::PlayerAircraft, Category::Pickup))
182 {
183 auto& player = static_cast<Aircraft&>(*pair.first);
184 auto& pickup = static_cast<Pickup&>(*pair.second);
185
186 // Apply pickup effect to player, destroy projectile
187 pickup.apply(player);
188 pickup.destroy();
189 }
190
191 else if (matchesCategories(pair, Category::EnemyAircraft, Category::AlliedProjectile)
192 || matchesCategories(pair, Category::PlayerAircraft, Category::EnemyProjectile))
193 {
194 auto& aircraft = static_cast<Aircraft&>(*pair.first);
195 auto& projectile = static_cast<Projectile&>(*pair.second);
196
197 // Apply projectile damage to aircraft, destroy projectile
198 aircraft.damage(projectile.getDamage());
199 projectile.destroy();
200 }
201 }

335void World::guideMissiles()
336{
337 // Setup command that stores all enemies in mActiveEnemies
338 Command enemyCollector;
339 enemyCollector.category = Category::EnemyAircraft;
340 enemyCollector.action = derivedAction<Aircraft>([this] (Aircraft& enemy, sf::Time)
341 {
342 if (!enemy.isDestroyed())
343 mActiveEnemies.push_back(&enemy);
344 });
345
346 // Setup command that guides all missiles to the enemy which is currently closest to the player
347 Command missileGuider;
348 missileGuider.category = Category::AlliedProjectile;
349 missileGuider.action = derivedAction<Projectile>([this] (Projectile& missile, sf::Time)
350 {
351 // Ignore unguided bullets
352 if (!missile.isGuided())
353 return;
354
355 float minDistance = std::numeric_limits<float>::max();
356 Aircraft* closestEnemy = nullptr;
357
358 // Find closest enemy
359 for(Aircraft* enemy: mActiveEnemies)
360 {
361 float enemyDistance = distance(missile, *enemy);
362
363 if (enemyDistance < minDistance)
364 {
365 closestEnemy = enemy;
366 minDistance = enemyDistance;
367 }
368 }
369
370 if (closestEnemy)
371 missile.guideTowards(closestEnemy->getWorldPosition());
372 });
373
374 // Push commands, reset active enemies
375 mCommandQueue.push(enemyCollector);
376 mCommandQueue.push(missileGuider);
377 mActiveEnemies.clear();
378}
