125bool MultiplayerGameState::update(sf::Time dt)
126{
127 // Connected to server: Handle all the network logic
128 if (mConnected)
129 {
130 mWorld.update(dt);
131
132 // Remove players whose aircrafts were destroyed
133 bool foundLocalPlane = false;
134 for (auto itr = mPlayers.begin(); itr != mPlayers.end(); )
135 {
136 // Check if there are no more local planes for remote clients
137 if (std::find(mLocalPlayerIdentifiers.begin(), mLocalPlayerIdentifiers.end(), itr->first) != mLocalPlayerIdentifiers.end())
138 {
139 foundLocalPlane = true;
140 }
141
142 if (!mWorld.getAircraft(itr->first))
143 {
144 itr = mPlayers.erase(itr);
145
146 // No more players left: Mission failed
147 if (mPlayers.empty())
148 requestStackPush(States::GameOver);
149 }
150 else
151 {
152 ++itr;
153 }
154 }
155
156 if (!foundLocalPlane && mGameStarted)
157 {
158 requestStackPush(States::GameOver);
159 }
160
161 // Only handle the realtime input if the window has focus and the game is unpaused
162 if (mActiveState && mHasFocus)
163 {
164 CommandQueue& commands = mWorld.getCommandQueue();
165 for (auto& pair : mPlayers)
166 pair.second->handleRealtimeInput(commands);
167 }
168
169 // Always handle the network input
170 CommandQueue& commands = mWorld.getCommandQueue();
171 for (auto& pair : mPlayers)
172 pair.second->handleRealtimeNetworkInput(commands);
173
174 // Handle messages from server that may have arrived
175 sf::Packet packet;
176 if (mSocket.receive(packet) == sf::Socket::Done)
177 {
178 mTimeSinceLastPacket = sf::seconds(0.f);
179 sf::Int32 packetType;
180 packet >> packetType;
181 handlePacket(packetType, packet);
182 }
183 else
184 {
185 // Check for timeout with the server
186 if (mTimeSinceLastPacket > mClientTimeout)
187 {
188 mConnected = false;
189
190 mFailedConnectionText.setString("Lost connection to server");
191 centerOrigin(mFailedConnectionText);
192
193 mFailedConnectionClock.restart();
194 }
195 }
196
197 updateBroadcastMessage(dt);
198
199 // Time counter for blinking 2nd player text
200 mPlayerInvitationTime += dt;
201 if (mPlayerInvitationTime > sf::seconds(1.f))
202 mPlayerInvitationTime = sf::Time::Zero;
203
204 // Events occurring in the game
205 GameActions::Action gameAction;
206 while (mWorld.pollGameAction(gameAction))
207 {
208 sf::Packet packet;
209 packet << static_cast<sf::Int32>(Client::GameEvent);
210 packet << static_cast<sf::Int32>(gameAction.type);
211 packet << gameAction.position.x;
212 packet << gameAction.position.y;
213
214 mSocket.send(packet);
215 }
216
217 // Regular position updates
218 if (mTickClock.getElapsedTime() > sf::seconds(1.f / 20.f))
219 {
220 sf::Packet positionUpdatePacket;
221 positionUpdatePacket << static_cast<sf::Int32>(Client::PositionUpdate);
222 positionUpdatePacket << static_cast<sf::Int32>(mLocalPlayerIdentifiers.size());
223
224 for (sf::Int32 identifier : mLocalPlayerIdentifiers)
225 {
226 if (Aircraft* aircraft = mWorld.getAircraft(identifier))
227 positionUpdatePacket << identifier << aircraft->getPosition().x << aircraft->getPosition().y << static_cast<sf::Int32>(aircraft->getHitpoints()) << static_cast<sf::Int32>(aircraft->getMissileAmmo());
228 }
229
230 mSocket.send(positionUpdatePacket);
231 mTickClock.restart();
232 }
233
234 mTimeSinceLastPacket += dt;
235 }
236
237 // Failed to connect and waited for more than 5 seconds: Back to menu
238 else if (mFailedConnectionClock.getElapsedTime() >= sf::seconds(5.f))
239 {
240 requestStateClear();
241 requestStackPush(States::Menu);
242 }
243
244 return true;
245}

255bool MultiplayerGameState::handleEvent(const sf::Event& event)
256{
257 // Game input handling
258 CommandQueue& commands = mWorld.getCommandQueue();
259
260 // Forward event to all players
261 for (auto& pair : mPlayers)
262 pair.second->handleEvent(event, commands);
263
264 if (event.type == sf::Event::KeyPressed)
265 {
266 // Enter pressed, add second player co-op (only if we are one player)
267 if (event.key.code == sf::Keyboard::Return && mLocalPlayerIdentifiers.size() == 1)
268 {
269 sf::Packet packet;
270 packet << static_cast<sf::Int32>(Client::RequestCoopPartner);
271
272 mSocket.send(packet);
273 }
274
275 // Escape pressed, trigger the pause screen
276 else if (event.key.code == sf::Keyboard::Escape)
277 {
278 disableAllRealtimeActions();
279 requestStackPush(States::NetworkPause);
280 }
281 }
282 else if (event.type == sf::Event::GainedFocus)
283 {
284 mHasFocus = true;
285 }
286 else if (event.type == sf::Event::LostFocus)
287 {
288 mHasFocus = false;
289 }
290
291 return true;
292}

316void MultiplayerGameState::handlePacket(sf::Int32 packetType, sf::Packet& packet)
317{
318 switch (packetType)
319 {
320 // Send message to all clients
321 case Server::BroadcastMessage:
322 {
323 std::string message;
324 packet >> message;
325 mBroadcasts.push_back(message);
326
327 // Just added first message, display immediately
328 if (mBroadcasts.size() == 1)
329 {
330 mBroadcastText.setString(mBroadcasts.front());
331 centerOrigin(mBroadcastText);
332 mBroadcastElapsedTime = sf::Time::Zero;
333 }
334 } break;
335
336 // Sent by the server to order to spawn player 1 airplane on connect
337 case Server::SpawnSelf:
338 {
339 sf::Int32 aircraftIdentifier;
340 sf::Vector2f aircraftPosition;
341 packet >> aircraftIdentifier >> aircraftPosition.x >> aircraftPosition.y;
342
343 Aircraft* aircraft = mWorld.addAircraft(aircraftIdentifier);
344 aircraft->setPosition(aircraftPosition);
345
346 mPlayers[aircraftIdentifier].reset(new Player(&mSocket, aircraftIdentifier, getContext().keys1));
347 mLocalPlayerIdentifiers.push_back(aircraftIdentifier);
348
349 mGameStarted = true;
350 } break;
351
352 //
353 case Server::PlayerConnect:
354 {
355 sf::Int32 aircraftIdentifier;
356 sf::Vector2f aircraftPosition;
357 packet >> aircraftIdentifier >> aircraftPosition.x >> aircraftPosition.y;
358
359 Aircraft* aircraft = mWorld.addAircraft(aircraftIdentifier);
360 aircraft->setPosition(aircraftPosition);
361
362 mPlayers[aircraftIdentifier].reset(new Player(&mSocket, aircraftIdentifier, nullptr));
363 } break;
364
365 //
366 case Server::PlayerDisconnect:
367 {
368 sf::Int32 aircraftIdentifier;
369 packet >> aircraftIdentifier;
370
371 mWorld.removeAircraft(aircraftIdentifier);
372 mPlayers.erase(aircraftIdentifier);
373 } break;
374
375 //
376 case Server::InitialState:
377 {
378 sf::Int32 aircraftCount;
379 float worldHeight, currentScroll;
380 packet >> worldHeight >> currentScroll;
381
382 mWorld.setWorldHeight(worldHeight);
383 mWorld.setCurrentBattleFieldPosition(currentScroll);
384
385 packet >> aircraftCount;
386 for (sf::Int32 i = 0; i < aircraftCount; ++i)
387 {
388 sf::Int32 aircraftIdentifier;
389 sf::Int32 hitpoints;
390 sf::Int32 missileAmmo;
391 sf::Vector2f aircraftPosition;
392 packet >> aircraftIdentifier >> aircraftPosition.x >> aircraftPosition.y >> hitpoints >> missileAmmo;
393
394 Aircraft* aircraft = mWorld.addAircraft(aircraftIdentifier);
395 aircraft->setPosition(aircraftPosition);
396 aircraft->setHitpoints(hitpoints);
397 aircraft->setMissileAmmo(missileAmmo);
398
399 mPlayers[aircraftIdentifier].reset(new Player(&mSocket, aircraftIdentifier, nullptr));
400 }
401 } break;
402
403 //
404 case Server::AcceptCoopPartner:
405 {
406 sf::Int32 aircraftIdentifier;
407 packet >> aircraftIdentifier;
408
409 mWorld.addAircraft(aircraftIdentifier);
410 mPlayers[aircraftIdentifier].reset(new Player(&mSocket, aircraftIdentifier, getContext().keys2));
411 mLocalPlayerIdentifiers.push_back(aircraftIdentifier);
412 } break;
413
414 // Player event (like missile fired) occurs
415 case Server::PlayerEvent:
416 {
417 sf::Int32 aircraftIdentifier;
418 sf::Int32 action;
419 packet >> aircraftIdentifier >> action;
420
421 auto itr = mPlayers.find(aircraftIdentifier);
422 if (itr != mPlayers.end())
423 itr->second->handleNetworkEvent(static_cast<Player::Action>(action), mWorld.getCommandQueue());
424 } break;
425
426 // Player's movement or fire keyboard state changes
427 case Server::PlayerRealtimeChange:
428 {
429 sf::Int32 aircraftIdentifier;
430 sf::Int32 action;
431 bool actionEnabled;
432 packet >> aircraftIdentifier >> action >> actionEnabled;
433
434 auto itr = mPlayers.find(aircraftIdentifier);
435 if (itr != mPlayers.end())
436 itr->second->handleNetworkRealtimeChange(static_cast<Player::Action>(action), actionEnabled);
437 } break;
438
439 // New enemy to be created
440 case Server::SpawnEnemy:
441 {
442 float height;
443 sf::Int32 type;
444 float relativeX;
445 packet >> type >> height >> relativeX;
446
447 mWorld.addEnemy(static_cast<Aircraft::Type>(type), relativeX, height);
448 mWorld.sortEnemies();
449 } break;
450
451 // Mission successfully completed
452 case Server::MissionSuccess:
453 {
454 requestStackPush(States::MissionSuccess);
455 } break;
456
457 // Pickup created
458 case Server::SpawnPickup:
459 {
460 sf::Int32 type;
461 sf::Vector2f position;
462 packet >> type >> position.x >> position.y;
463
464 mWorld.createPickup(position, static_cast<Pickup::Type>(type));
465 } break;
466
467 //
468 case Server::UpdateClientState:
469 {
470 float currentWorldPosition;
471 sf::Int32 aircraftCount;
472 packet >> currentWorldPosition >> aircraftCount;
473
474 float currentViewPosition = mWorld.getViewBounds().top + mWorld.getViewBounds().height;
475
476 // Set the world's scroll compensation according to whether the view is behind or too advanced
477 mWorld.setWorldScrollCompensation(currentViewPosition / currentWorldPosition);
478
479 for (sf::Int32 i = 0; i < aircraftCount; ++i)
480 {
481 sf::Vector2f aircraftPosition;
482 sf::Int32 aircraftIdentifier;
483 packet >> aircraftIdentifier >> aircraftPosition.x >> aircraftPosition.y;
484
485 Aircraft* aircraft = mWorld.getAircraft(aircraftIdentifier);
486 bool isLocalPlane = std::find(mLocalPlayerIdentifiers.begin(), mLocalPlayerIdentifiers.end(), aircraftIdentifier) != mLocalPlayerIdentifiers.end();
487 if (aircraft && !isLocalPlane)
488 {
489 sf::Vector2f interpolatedPosition = aircraft->getPosition() + (aircraftPosition - aircraft->getPosition()) * 0.1f;
490 aircraft->setPosition(interpolatedPosition);
491 }
492 }
493 } break;
494 }
