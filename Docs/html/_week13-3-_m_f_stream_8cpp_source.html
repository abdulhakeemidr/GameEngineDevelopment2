193 //
194 // Initialize XAudio2
195 //
196 HRESULT hr = CoInitializeEx(nullptr, COINIT_MULTITHREADED);
197 if (FAILED(hr))
198 {
199 wprintf(L"Failed to init COM: %#X\n", hr);
200 return 0;
201 }
202
203#ifdef USING_XAUDIO2_7_DIRECTX

205#ifdef _DEBUG
206 HMODULE mXAudioDLL = LoadLibraryExW(L"XAudioD2_7.DLL", nullptr, 0x00000800 /* LOAD_LIBRARY_SEARCH_SYSTEM32 */);
207#else
208 HMODULE mXAudioDLL = LoadLibraryExW(L"XAudio2_7.DLL", nullptr, 0x00000800 /* LOAD_LIBRARY_SEARCH_SYSTEM32 */);
209#endif
210 if (!mXAudioDLL)
211 {
212 wprintf(L"Failed to find XAudio 2.7 DLL");
213 CoUninitialize();
214 return 0;
215 }
216#endif
217
218 UINT32 flags = 0;
219#if defined(USING_XAUDIO2_7_DIRECTX) && defined(_DEBUG)
220 flags |= XAUDIO2_DEBUG_ENGINE;
221#endif
222 ComPtr<IXAudio2> pXAudio2;
223 hr = XAudio2Create(pXAudio2.GetAddressOf(), flags);
224 if (FAILED(hr))
225 {
226 wprintf(L"Failed to init XAudio2 engine: %#X\n", hr);
227 CoUninitialize();
228 return 0;
229 }
230
231#if !defined(USING_XAUDIO2_7_DIRECTX) && defined(_DEBUG)
232 // To see the trace output, you need to view ETW logs for this application:
233 // Go to Control Panel, Administrative Tools, Event Viewer.
234 // View->Show Analytic and Debug Logs.
235 // Applications and Services Logs / Microsoft / Windows / XAudio2.
236 // Right click on Microsoft Windows XAudio2 debug logging, Properties, then Enable Logging, and hit OK
237 XAUDIO2_DEBUG_CONFIGURATION debug = { 0 };
238 debug.TraceMask = XAUDIO2_LOG_ERRORS | XAUDIO2_LOG_WARNINGS;
239 debug.BreakMask = XAUDIO2_LOG_ERRORS;
240 pXAudio2->SetDebugConfiguration(&debug, 0);
241#endif
242
243 //
244 // Create a mastering voice
245 //
246 IXAudio2MasteringVoice* pMasteringVoice = nullptr;
247
248 if (FAILED(hr = pXAudio2->CreateMasteringVoice(&pMasteringVoice)))
249 {
250 wprintf(L"Failed creating mastering voice: %#X\n", hr);
251 pXAudio2.Reset();
252 CoUninitialize();
253 return 0;
254 }
255
256 //
257 // Find our media file
258 //
259 WCHAR mediaFile[MAX_PATH];
260
261 if (FAILED(hr = FindMediaFileCch(mediaFile, MAX_PATH, L"Media\\Wavs\\becky.wma")))
262 {
263 wprintf(L"Failed to find media file (%#X)\n", hr);
264 pXAudio2.Reset();
265 CoUninitialize();
266 return 0;
267 }
268
269 //
270 // Start up Media Foundation
271 //
272 hr = MFStartup(MF_VERSION);
273 if (FAILED(hr))
274 {
275 wprintf(L"Failed to initialize Media Foundation (%#X)\n", hr);
276 pXAudio2.Reset();
277 CoUninitialize();
278 return 0;
279 }
280
281 //
282 // Create MF reader for our media file
283 //
284 SourceReaderContext readerContext;
285 ComPtr<IMFSourceReader> reader;
286 WAVEFORMATEX wfx;
287 hr = CreateMFReader(mediaFile, &readerContext, reader.GetAddressOf(), &wfx, sizeof(wfx));
288 if (FAILED(hr))
289 {
290 wprintf(L"Failed to create media reader (%#X)\n", hr);
291 pXAudio2.Reset();
292 CoUninitialize();
293 return 0;
294 }
295
296 //
297 // Create the source voice
298 //
299 StreamingVoiceContext voiceContext;
300
301 IXAudio2SourceVoice* pSourceVoice;
302 hr = pXAudio2->CreateSourceVoice(&pSourceVoice, &wfx, 0, 1.0f, &voiceContext);
303 if (FAILED(hr))
304 {
305 wprintf(L"Error %#X creating source voice\n", hr);
306 pXAudio2.Reset();
307 CoUninitialize();
308 return 0;
309 }
310 pSourceVoice->Start(0, 0);
311
312 //
313 // This loop continously updates the audio stream. Within an application, this should be handled
314 // on a worker thread rather than the main thread.
315 //
316 wprintf(L"Press <ESC> to exit.\n");
317 wprintf(L"Now playing %s", mediaFile);
318
319 DWORD currentStreamBuffer = 0;
320
321 hr = S_OK;
322
323 size_t bufferSize[MAX_BUFFER_COUNT] = { 0 };
324 std::unique_ptr<uint8_t[]> buffers[MAX_BUFFER_COUNT];
325
326 for (;;)
327 {
328 wprintf(L".");
329
330 if (GetAsyncKeyState(VK_ESCAPE))
331 {
332 while (GetAsyncKeyState(VK_ESCAPE))
333 Sleep(10);
334
335 break;
336 }
337
338 DWORD flags = 0;
339 hr = reader->ReadSample(MF_SOURCE_READER_FIRST_AUDIO_STREAM, 0, nullptr, nullptr, nullptr, nullptr);
340 if (FAILED(hr))
341 break;
342
343 WaitForSingleObject(readerContext.hReadSample, INFINITE);
344
345 if (readerContext.endOfStream)
346 {
347 wprintf(L"restart streaming..");
348 readerContext.Restart();
349
350 // Restart loop
351 PROPVARIANT var = { 0 };
352 var.vt = VT_I8;
353
354 hr = reader->SetCurrentPosition(GUID_NULL, var);
355 if (SUCCEEDED(hr))
356 continue;
357 else
358 break;
359 }
360
361 ComPtr<IMFMediaBuffer> mediaBuffer;
362 hr = readerContext.sample->ConvertToContiguousBuffer(mediaBuffer.GetAddressOf());
363 if (FAILED(hr))
364 return hr;
365
366 BYTE* audioData = nullptr;
367 DWORD sampleBufferLength = 0;
368
369 hr = mediaBuffer->Lock(&audioData, nullptr, &sampleBufferLength);
370 if (FAILED(hr))
371 return hr;
372
373 if (bufferSize[currentStreamBuffer] < sampleBufferLength)
374 {
375 buffers[currentStreamBuffer].reset(new uint8_t[sampleBufferLength]);
376 bufferSize[currentStreamBuffer] = sampleBufferLength;
377 }
378
379 memcpy_s(buffers[currentStreamBuffer].get(), sampleBufferLength, audioData, sampleBufferLength);
380
381 hr = mediaBuffer->Unlock();
382 if (FAILED(hr))
383 return hr;
384
385 //
386 // Wait to se if our XAudio2 source voice has played enough data for us to give
387 // it another buffer full of audio. We'd like to keep no more than MAX_BUFFER_COUNT - 1
388 // buffers on the queue, so that one buffer is always free for the MF streamer
389 //
390 XAUDIO2_VOICE_STATE state;
391 for (;; )
392 {
393 pSourceVoice->GetState(&state);
394 if (state.BuffersQueued < MAX_BUFFER_COUNT - 1)
395 break;
396
397 WaitForSingleObject(voiceContext.hBufferEndEvent, INFINITE);
398 }
399
400 XAUDIO2_BUFFER buf = { 0 };
401 buf.AudioBytes = sampleBufferLength;
402 buf.pAudioData = buffers[currentStreamBuffer].get();
403 pSourceVoice->SubmitSourceBuffer(&buf);
404
405 currentStreamBuffer++;
406 currentStreamBuffer %= MAX_BUFFER_COUNT;
407 }
408
409 if (FAILED(hr))
410 {
411 wprintf(L"\nError %#X during playback!\n", hr);
412 }
413
414 reader->Flush(MF_SOURCE_READER_FIRST_AUDIO_STREAM);
415
416 //
417 // Cleanup XAudio2
418 //
419
420 // All XAudio2 interfaces are released when the engine is destroyed, but being tidy
421 pSourceVoice->DestroyVoice();
422 pMasteringVoice->DestroyVoice();
423
424 //
425 // Cleanup Media Foundation
426 //
427 reader = nullptr;
428 MFShutdown();
429
430 pXAudio2.Reset();
431
432#ifdef USING_XAUDIO2_7_DIRECTX
433 if (mXAudioDLL)
434 FreeLibrary(mXAudioDLL);
435#endif
436
437 CoUninitialize();
438}

442// Helper for setting up the MF source reader
443//--------------------------------------------------------------------------------------
444HRESULT CreateMFReader(_In_z_ const WCHAR* mediaFile, _In_ IUnknown* async, _Outptr_ IMFSourceReader** reader, _Out_ WAVEFORMATEX* wfx, _In_ size_t maxwfx)
445{
446 if (!mediaFile || !reader || !wfx)
447 return E_INVALIDARG;
448
449 HRESULT hr;
450 ComPtr<IMFAttributes> attr;
451#if (_WIN32_WINNT >= 0x0602 /*_WIN32_WINNT_WIN8*/)
452 hr = MFCreateAttributes(attr.GetAddressOf(), 2);
453 if (FAILED(hr))
454 return hr;
455
456 hr = attr->SetUINT32(MF_LOW_LATENCY, TRUE);
457#else
458 hr = MFCreateAttributes(attr.GetAddressOf(), 1);
459#endif
460 if (FAILED(hr))
461 return hr;
462
463 hr = attr->SetUnknown(MF_SOURCE_READER_ASYNC_CALLBACK, async);
464 if (FAILED(hr))
465 return hr;
466
467 hr = MFCreateSourceReaderFromURL(mediaFile, attr.Get(), reader);
468 if (FAILED(hr))
469 return hr;
470
471 //
472 // Make the output from Media Foundation PCM so XAudio2 can consume it
473 //
474
475 ComPtr<IMFMediaType> mediaType;
476 hr = MFCreateMediaType(mediaType.GetAddressOf());
477 if (FAILED(hr))
478 return hr;
479
480 hr = mediaType->SetGUID(MF_MT_MAJOR_TYPE, MFMediaType_Audio);
481 if (FAILED(hr))
482 return hr;
483
484 hr = mediaType->SetGUID(MF_MT_SUBTYPE, MFAudioFormat_PCM);
485 if (FAILED(hr))
486 return hr;
487
488 hr = (*reader)->SetCurrentMediaType(MF_SOURCE_READER_FIRST_AUDIO_STREAM, 0, mediaType.Get());
489 if (FAILED(hr))
490 return hr;
491
492 //
493 // Get the wave format
494 //
495
496 ComPtr<IMFMediaType> outputMediaType;
497 hr = (*reader)->GetCurrentMediaType(MF_SOURCE_READER_FIRST_AUDIO_STREAM, outputMediaType.GetAddressOf());
498 if (FAILED(hr))
499 return hr;
500
501 UINT32 waveFormatSize = 0;
502 WAVEFORMATEX* waveFormat = nullptr;
503 hr = MFCreateWaveFormatExFromMFMediaType(outputMediaType.Get(), &waveFormat, &waveFormatSize);
504 if (FAILED(hr))
505 return hr;
506
507 memcpy_s(wfx, maxwfx, waveFormat, waveFormatSize);
508 CoTaskMemFree(waveFormat);
509
510 return S_OK;
511}

518HRESULT FindMediaFileCch(WCHAR* strDestPath, int cchDest, LPCWSTR strFilename)
519{
520 bool bFound = false;
521
522 if (!strFilename || strFilename[0] == 0 || !strDestPath || cchDest < 10)
523 return E_INVALIDARG;
524
525 // Get the exe name, and exe path
526 WCHAR strExePath[MAX_PATH] = { 0 };
527 WCHAR strExeName[MAX_PATH] = { 0 };
528 WCHAR* strLastSlash = nullptr;
529 GetModuleFileName(nullptr, strExePath, MAX_PATH);
530 strExePath[MAX_PATH - 1] = 0;
531 strLastSlash = wcsrchr(strExePath, TEXT('\\'));
532 if (strLastSlash)
533 {
534 wcscpy_s(strExeName, MAX_PATH, &strLastSlash[1]);
535
536 // Chop the exe name from the exe path
537 *strLastSlash = 0;
538
539 // Chop the .exe from the exe name
540 strLastSlash = wcsrchr(strExeName, TEXT('.'));
541 if (strLastSlash)
542 *strLastSlash = 0;
543 }
544
545 wcscpy_s(strDestPath, cchDest, strFilename);
546 if (GetFileAttributes(strDestPath) != 0xFFFFFFFF)
547 return S_OK;
548
549 // Search all parent directories starting at .\ and using strFilename as the leaf name
550 WCHAR strLeafName[MAX_PATH] = { 0 };
551 wcscpy_s(strLeafName, MAX_PATH, strFilename);
552
553 WCHAR strFullPath[MAX_PATH] = { 0 };
554 WCHAR strFullFileName[MAX_PATH] = { 0 };
555 WCHAR strSearch[MAX_PATH] = { 0 };
556 WCHAR* strFilePart = nullptr;
557
558 GetFullPathName(L".", MAX_PATH, strFullPath, &strFilePart);
559 if (!strFilePart)
560 return E_FAIL;
561
562 while (strFilePart && *strFilePart != '\0')
563 {
564 swprintf_s(strFullFileName, MAX_PATH, L"%s\\%s", strFullPath, strLeafName);
565 if (GetFileAttributes(strFullFileName) != 0xFFFFFFFF)
566 {
567 wcscpy_s(strDestPath, cchDest, strFullFileName);
568 bFound = true;
569 break;
570 }
571
572 swprintf_s(strFullFileName, MAX_PATH, L"%s\\%s\\%s", strFullPath, strExeName, strLeafName);
573 if (GetFileAttributes(strFullFileName) != 0xFFFFFFFF)
574 {
575 wcscpy_s(strDestPath, cchDest, strFullFileName);
576 bFound = true;
577 break;
578 }
579
580 swprintf_s(strSearch, MAX_PATH, L"%s\\..", strFullPath);
581 GetFullPathName(strSearch, MAX_PATH, strFullPath, &strFilePart);
582 }
583 if (bFound)
584 return S_OK;
585
586 // On failure, return the file as the path but also return an error code
587 wcscpy_s(strDestPath, cchDest, strFilename);
588
589 return HRESULT_FROM_WIN32(ERROR_FILE_NOT_FOUND);
590}
