215void Aircraft::updateMovementPattern(sf::Time dt)
216{
217 // Enemy airplane: Movement pattern
218 const std::vector<Direction>& directions = Table[mType].directions;
219 if (!directions.empty())
220 {
221 // Moved long enough in current direction: Change direction
222 if (mTravelledDistance > directions[mDirectionIndex].distance)
223 {
224 mDirectionIndex = (mDirectionIndex + 1) % directions.size();
225 mTravelledDistance = 0.f;
226 }
227
228 // Compute velocity from direction
229 float radians = toRadian(directions[mDirectionIndex].angle + 90.f);
230 float vx = getMaxSpeed() * std::cos(radians);
231 float vy = getMaxSpeed() * std::sin(radians);
232
233 setVelocity(vx, vy);
234
235 mTravelledDistance += getMaxSpeed() * dt.asSeconds();
236 }
237}

247void Aircraft::checkProjectileLaunch(sf::Time dt, CommandQueue& commands)
248{
249 // Enemies try to fire all the time

252
253 // Check for automatic gunfire, allow only in intervals
254 if (mIsFiring && mFireCountdown <= sf::Time::Zero)

256 // Interval expired: We can fire a new bullet

261#pragma endregion

265 mIsFiring = false;
