\doxysubsection{Detailed Description}
Encapsulate an IPv4 network address. 

\mbox{\hyperlink{classsf_1_1_ip_address}{sf\+::\+Ip\+Address}} is a utility class for manipulating network addresses.

It provides a set a implicit constructors and conversion functions to easily build or transform an IP address from/to various representations.

Usage example\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_ip_address}{sf::IpAddress}} a0;                                     \textcolor{comment}{// an invalid address}}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_ip_address}{sf::IpAddress}} a1 = \mbox{\hyperlink{classsf_1_1_ip_address_af95b0353e632372b5266d6227bc0e3ec}{sf::IpAddress::None}};               \textcolor{comment}{// an invalid address (same as a0)}}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_ip_address}{sf::IpAddress}} a2(\textcolor{stringliteral}{"{}127.0.0.1"{}});                        \textcolor{comment}{// the local host address}}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_ip_address}{sf::IpAddress}} a3 = \mbox{\hyperlink{classsf_1_1_ip_address_aefb978156901e01e2d9642815be10abe}{sf::IpAddress::Broadcast}};          \textcolor{comment}{// the broadcast address}}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_ip_address}{sf::IpAddress}} a4(192, 168, 1, 56);                    \textcolor{comment}{// a local address}}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_ip_address}{sf::IpAddress}} a5(\textcolor{stringliteral}{"{}my\_computer"{}});                      \textcolor{comment}{// a local address created from a network name}}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_ip_address}{sf::IpAddress}} a6(\textcolor{stringliteral}{"{}89.54.1.169"{}});                      \textcolor{comment}{// a distant address}}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_ip_address}{sf::IpAddress}} a7(\textcolor{stringliteral}{"{}www.google.com"{}});                   \textcolor{comment}{// a distant address created from a network name}}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_ip_address}{sf::IpAddress}} a8 = \mbox{\hyperlink{classsf_1_1_ip_address_a4c31622ad87edca48adbb8e8ed00ee4a}{sf::IpAddress::getLocalAddress}}();  \textcolor{comment}{// my address on the local network}}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1_ip_address}{sf::IpAddress}} a9 = \mbox{\hyperlink{classsf_1_1_ip_address_a5c5cbf67e4aacf23c24f2ad991df4c55}{sf::IpAddress::getPublicAddress}}(); \textcolor{comment}{// my address on the internet}}

\end{DoxyCode}


Note that \mbox{\hyperlink{classsf_1_1_ip_address}{sf\+::\+Ip\+Address}} currently doesn\textquotesingle{}t support IPv6 nor other types of network addresses.
